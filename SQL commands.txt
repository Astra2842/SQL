create table *название таблицы* (столбцы и параметры столбцов) - создать таблицу;
параметры столбцов: 
	- serial (для postgresql) - порядковый номер;
	- varchar (цифра) - количество символов в ячейке; 
	- unique - ячейки в столбце имеют уникальные значения
	- not null - не ноль
	
insert into *название таблицы*	(*название столбца*)
values (*значение ячейки*) - наполнение таблицы значениями

alter table *название таблицы* 
add column *название *столбца* *параметр столбца* - добавление столбца

alter table *название таблицы* 
add column *название *столбца* *параметр столбца* to *новое название* - переименование столбца.

alter table *название таблицы*
	drop column *название *столбца*; - удаление столбца
	
	
update *название таблицы*
set *название *столбца 1* = значение
where *название *столбца 2*=значение; - отредактировать ячейку


delete from *название таблицы*
where *название *столбца*=значение; удаление строки

ALTER TABLE [tbl_name] ALTER COLUMN [col_name] [DATA_TYPE] - изменение типа столбца

select * *название таблицы* - вывести всю таблицу;
